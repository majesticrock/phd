\section{Model}\label{sec:model}

We will focus on a rather general Hamiltonian at zero temperature given by
\begin{equation}
\label{eqn:base_hamiltonian}
  H = H_\mathrm{kin} + H_\mathrm{Ph} + H_\mathrm{C} + H_\mathrm{BG},
\end{equation}
where $H_\mathrm{kin}$ describes the kinetic part,
$H_\mathrm{Ph}$ describes an effective electron-electron interaction mediated by phonons, 
$H_\mathrm{C}$ describes the repulsive Coulomb interaction between electrons,
and $H_\mathrm{BG}$ describes the electronic interaction between the electrons and the atomic nuclei.
The individual terms are given by
\bs
\begin{align}
  H_\mathrm{Kin} &= \sum_{\vec{k} \sigma} \left( \frac{\hbar^2 k^2}{2 m_e} - \Ef \right) n_{k\sigma} \\
  H_\mathrm{Ph} &= - \frac{1}{V} \sum_{\vec{k} \vec{k'}} g(k,k') \opSCd{k} \opSC{k'} \\
  H_\mathrm{C} &= \frac{e^2}{2V \epsilon_0} \sum_{\sigma \sigma'} \sum_{\vec{k} \vec{k}' \vec{q}} \frac{1}{q^2 + k_s^2} \opd{k}{\sigma} \opd{k}{'\sigma} c_{\vec{k}'-\vec{q} \sigma'} c_{\vec{k}-\vec{q} \sigma} \\
  H_\mathrm{BG} &= - \frac{e^2}{2 \epsilon_0 k_s^2} \sum_{ \vec{k} \sigma } \rho   c_{ \vec{k}, \sigma }^\dagger  c_{ \vec{k}, \sigma }.
\end{align}
\es
Here, $V$ is the system's volume, $\Ef$ the Fermi energy, $\hbar$ the reduced Planck's constant, $m_e$ the electron's mass, 
$e$ its charge, $\epsilon_0$ the vacuum permitivity, and $k_s$ the Thomas-Fermi screening wavevector.
The operator $c_{ \vec{k}, \sigma }^{(\dagger)}$ annihilates (creates) an electron with momentum $\vec{k}$ and spin $\sigma$.

The phononic interaction is given by \cite{krull2012}
\begin{equation}
\label{eqn:phonon_int}
  g(k, k') = G \Theta( \debye - |\xi(k) - \xi(k')| ),
\end{equation}
where $G > 0$ is the interaction strength, $\debye$ the Debye frequency, $\Theta(k)$ the Heaviside function,
and $\xi(k)$ describes the single-particle dispersion, i.e.,
$\xi(k) = \hbar^2 k^2 / (2m_e) + \epsilon_\mathrm{Fock}(k) - \Ef$.
The Fock energy $\epsilon_\mathrm{F}$ originates from the mean-field treatment of $H_\mathrm{C}$, see \eqref{eqn:fock_coulomb} below.
This interaction can be derived using continuous unitary transformation and is identical to the commonly known result by Fr\"ohlich for real processes \cite{frohlich1997,krull2012}.

The quantity 
\begin{equation}
  \rho \coloneqq \frac{1}{V} \sum_{\vec{k'} \sigma'} \expec{c_{ \vec{k}', \sigma' }^\dagger  c_{ \vec{k}', \sigma' }}
\end{equation}
describes the atomic nuclei qualitatively as a constant positive background charge.
It exactly cancels with the divergent Hartree contribution in $H_\mathrm{C}$.

Throughout this article, we fix the Fermi wavevector to $\hbar k_\mathrm{F} / \sqrt{m_e} = 4.25 \sqrt{\mathrm{eV}}$.
Furthermore, we define the dimensionless parameter 
\begin{equation}
  g \coloneqq \rho_\mathrm{F} G = G \frac{m_e \kf}{2 \pi^2 \hbar^2},
\end{equation}
where $\rho_\mathrm{F}$ is the density of states of the non-interacting system at the Fermi edge.
In the same manner, we define a dimensionless scaling $\lambda$ for the screening wavevector
\begin{equation}
  k_s \coloneqq \lambda \sqrt{\frac{e^2 m_e}{3 \pi^2 \epsilon_0} \kf}.
\end{equation}
The square-root expression is an estimate for the Thomas-Fermi wavevector in real materials provided by Ref. \cite{rickayzen80}.
Lastly, we will define the following operators for brevity
\begin{equation}
  f_{\vec{k}} \coloneqq \opSC{k},\qquad n_{\vec{k}\sigma} \coloneqq \opd{k}{\sigma} \ope{k}{\sigma}.
\end{equation}

\section{The superconducting gap function}
\label{sec:mean_field}

\subsection{General considerations}

Computing quanties within the model \eqref{eqn:base_hamiltonian} requires some treatment of the interaction terms.
Within this article, we use the approach described in Ref. \cite{althuser2024}.
Therefore, we begin by formulating a mean-field decoupling of the interaction terms, which will provide us with access to various expectation values.

The phononic contribution boils down to a single non-vanishing term
\begin{equation}
  \tilde{H}_\mathrm{Ph} = \sum_{\vec{k}} \Delta_\mathrm{Ph} (k) f_{\vec{k}}^\dagger + \mathrm{H.c.},
\end{equation}
where the phononic contribution the the superconducting gap function is given by
\begin{align}
\label{eqn:delta_phonon}
  \Delta_\mathrm{Ph} (k) &= \frac{1}{V} \sum_{\vec{k'}} g(k,k')  \expec{f_k} \nonumber \\
    &= \frac{1}{2 \pi^2} \int_0^\infty \mathrm{d} k' k'^2 g(k, k') \expec{ f_k }.
\end{align}

Repeating the same calculation for $H_\mathrm{C}$ yields
\begin{align}
  \tilde{H}_\mathrm{C} &= - \sum_{\vec{k}} \Delta_\mathrm{C} (k) f_{\vec{k}} + \mathrm{H.c.} \nonumber \\
    &+ \sum_{\vec{k} \sigma} (\epsilon_\mathrm{Fock}(k) + \epsilon_\mathrm{C}(k)) n_{\vec{k} \sigma}.
\end{align}
The additional contribution to the gap function is given by
\begin{equation}
\label{eqn:delta_coulomb}
  \Delta_\mathrm{C} (k) = \frac{e^2}{8 \pi^2 \epsilon_0 k} \int_0^\infty \mathrm{d} q \expec{f_{q}^\dagger} q \ln \left( \frac{k_s^2 + (q + k)^2}{k_s^2 + (q - k)^2} \right).
\end{equation}
Note, that this term has the opposite sign of \eqref{eqn:delta_phonon}.
The Fock energy reads
\begin{align}
\label{eqn:fock_coulomb}
  \epsilon_\mathrm{Fock} (k) &= - \frac{e^2}{4 \pi^2 \epsilon_0} \kf \bigg[1 \nonumber \\
  &+ \frac{k_s}{\kf } \left( \arctan \left( \frac{k - \kf}{k_s} \right) - \arctan \left( \frac{k + \kf}{k_s} \right) \right) \nonumber \\
  &+ \frac{\kf^2 - k^2 + k_s^2}{2 k \kf} \ln \left( \frac{k_s^2 + (\kf + k)^2}{k_s^2 + (\kf - k)^2} \right) \bigg].
\end{align}
The above expression assumes that $\expec{n_{\vec{k} \sigma}} = \Theta (\kf - k)$.
Naturally, this is not entirely accurate within the superconducting phase.
Therefore, we additionally find a small correction the Fock energy
\begin{equation}
\label{eqn:fock_corr}
  \epsilon_\mathrm{C}(k) = \frac{e^2}{8 \pi^2 \epsilon_0 k} \int_0^\infty \mathrm{d} q \delta n_{q} q \ln \left( \frac{k_s^2 + (q + k)^2}{k_s^2 + (q - k)^2} \right),
\end{equation}
where we defined $\delta n_{k} \coloneqq \Theta (\kf - k) - \expec{n_{k\sigma}}$.
This quantity is usually in the order of a few microelectronvolts.

For our numerics, we take notice that all of the quantities above merely depend on the absolute value $k = |\vec{k}|$.
Therefore, we discretize $k$ using $N_k$ individual points. 
Due to the long-range nature of the Coulomb interaction,
we need to stretch our discretization across a large range of $k$, theoretically even from $0$ to $\infty$,
but in practice $k \in [0, 2 \kf]$ produces good results.
Nevertheless, the main features can be observed around the Fermi edge causing a equidistant discretization to be insufficient.
To combat this, we choose a fine mesh around the Fermi edge that utilizes $N_k / 2$ individual points and is cut off at $\kf \pm x_\mathrm{cut} \debye / \kf$.
After this cutoff, the discretization is much coarser with $N_k / 4$ additional discretization points in each direction.
Choosing $x_\mathrm{cut}$ is somewhat arbitrary.
If the gap is small, a small value is beneficial as to resolve the relevant part of the gap function in more detail.
If the gap is large, however, we require a larger $x_\mathrm{cut}$ so that we include the entire relevant part in the first place.
In practice, $x_\mathrm{cut} = 20$ produces good results and will be used throughout the article. \textcolor{red}{Teste noch etwas rum, ob es eine bessere Wahl gibt.}

\subsection{Discussion of the numerical results}

\begin{figure}
  \centering
  \includegraphics[width=.48\textwidth]{plots/gap_theta.pdf}
  \caption{The superconducting gap function without Coulomb interactions. 
  The blue solid line shows the result for the exact interaction \eqref{eqn:phonon_int}
  while the orange dashed line depicts the data for the approximation \eqref{eqn:approx_g}.
  For both cases, we set $N_k = 4000$, $g=0.5$, and $\debye = \SI{10}{meV}$.}
  \label{fig:gap_theta}
\end{figure}

Let us begin by omitting the Coulomb interactions, i.e., setting $e=0$.
Then, the gap function naturally only includes the phononic contribution.
Oftentimes, the effective electron-electron interaction is assumed to be constant in a small region around the Fermi edge [einige Beispielzitate].
It is then described by
\begin{equation}
\label{eqn:approx_g}
  g(k, k') \coloneqq G \Theta(\debye - |\xi(k)|) \Theta(\debye - |\xi(k')|).
\end{equation}
This case shall suit as a base line for comparision's sake now.
An exemplary gap function for both, this approximate interaction and the exact one \eqref{eqn:phonon_int}, is depicted in \autoref{fig:gap_theta}.
We set $N_k = 4000$, $g=0.5$, and $\debye = \SI{10}{meV}$, but the gap function itself is qualitatively generic.
Increasing $g$ increases its magnitude and increasing $\debye$ increases its width.
As to be expected, both functions are restricted to a small region around the Fermi edge.
The latter, however, is slightly wider but exhibits a smaller magnitude.
All in all, the approximation is certainly justified for a qualitative description of the gap function.
Nonetheless, we will continue using the exact form for this article.

\begin{figure}
  \centering
  \includegraphics[width=.48\textwidth]{plots/gap_coulomb.pdf}
  \caption{The superconducting gap function with Coulomb interactions. 
  The blue and orange dashed lines depict the phononic \eqref{eqn:delta_phonon} and Coulomb \eqref{eqn:delta_coulomb} contributions to the gap function,
  which itself is represented by the black solid line.
  The green dotted line shows the correction to the Fock energy \eqref{eqn:fock_corr}.
  We used $\lambda = 10^{-4}$ for panel (a) and $\lambda = 1$ for panel(b). Note the difference in energy scales between the two panels.
  For both cases, we set $N_k = 20000$, $g=0.8$, and $\debye = \SI{10}{meV}$.}
  \label{fig:gap_coulomb}
\end{figure}

\begin{figure}
  \centering
  \includegraphics[width=.48\textwidth]{plots/gap_behavior.pdf}
  \caption{Plot of the maximum gap value $\maxGap$ depending on the screening $\lambda$.
  The former is scaled logarithmically. The phononic interaction strength is set to $g=0.5$ (blue solid line) and $g=0.7$ (orange dashed line).
  Otherwise, the parameters are identical to the ones in \autoref{fig:gap_coulomb}}
  \label{fig:gap_behavior}
\end{figure}

Next, we activate the Coulomb interaction again which results in considerable changes to the gap function.
A plot of two exemplary gap functions can be seen in \autoref{fig:gap_coulomb}.
Here, we used $N_k = 20000$, $g=0.8$, and $\debye = \SI{10}{meV}$. 
In the top panel, the screening is set to $\lambda = 10^{-4}$ while the bottom panel has it set to $\lambda = 1$.
Again, the functions are qualitatively generic with regard to change in $g$ and $\debye$.
Naturally, increasing the screening also increases the gap's magnitude.
Also, as mentioned before, the correction to the Fock energy $\epsilon_C$ is neglectably small.
\newline
The most striking new feature is that the gap function, indepedent of the screening, stretches to all $k$ values.
While its main contribution is still confined to a small region around the edge, 
it approach a constant value for $k \to 0$ and follows a $1/k^2$ behavior as $k \to \infty$. 
\textcolor{red}{Das kann man analytisch zeigen - wollen/sollen wir das hier tun?}
Furthermore, due to the repulsive nature of the Coulomb interaction, the gap function has a radial node when said interaction starts dominating over the phononic one.
\textcolor{red}{TODO: etwas Literatureinordnung gem\"a\ss{} der Quellen von Ilya.}
At this point, it should be mentioned that the complex phase of the gap function can still be fixed to be real.
We repeated the calculation allowing for an arbitrary complex phase, but the results are essentially identical:
Instead of changing sign at the nodal points, the complex phase of the gap function jumps by $\pi$.
\newline
Considering a small screening, i.e., $\lambda = 10^{-4}$ in panel (a), results in a rather narrow valley in $\Delta_\mathrm{C}$ around the Fermi edge.
It is noteworthy, that this result is essentially identical to the one obtained by omiting the screening entirely because this feature originates from integrating over a logarithmic singularity.
While this part of our calculations can still be easily performed without a screening, the later parts require atleast a small one as to avoid numerically dividing by $0$.
\newline
The aforementioned feature disappears entirely if we use the screening wavevector presented in Ref. \cite{rickayzen80}, i.e., $\lambda = 1$ in panel (b).
In this case, the contribution of $\Delta_\mathrm{C}$ around the Fermi edge is essentially a constant.
This corroborates the qualitative statements of the commonly used result by Morel and Anderson \cite{morel1962}
that a screened Coulomb interaction essentially boils down to a constant pseudo-potential around the Fermi edge \cite{sigrist2005,kostrzewa2018,simonato2023}.

\autoref{fig:gap_behavior} shows the maximum gap value $\maxGap$ depending on the screening $\lambda$.
Naturally, it grows as the screening is increased, but it does not seem to follow any elementary function.
Expectatly, the function approaches a constant value for both cases $\lambda \to 0$ and $\lambda \to \infty$.
The general behavior is generic across various values of the phononic coupling strength $g$.

\section{Collective excitations}

\subsection{General considerations}

\bullet use ieom as described in \cite{althuser2024}.

\bullet Similar basis idea:
\begin{equation}
  B_k \coloneqq \frac{1}{\sqrt{V}} \sum_{\vec{q}} h(k - |\vec{q}|) O_{\vec{q}}.
\end{equation}
$O_{\vec{q}}$ represents $f_{\vec{q}}$ and $n_{\vec{k}\up} +  n_{-\vec{k}\down}$

\bullet We will investigate the Higgs and the phase modes, given by
\begin{subequations}
\begin{align}
    A_\mathrm{Higgs} &\coloneqq \frac{1}{\sqrt{V}} \sum_{\vec{k}} \left( f_{\vec{k}}^\dagger + f_{\vec{k}} \right), \\
    A_\mathrm{Phase} &\coloneqq \frac{i}{\sqrt{V}} \sum_{\vec{k}} \left( f_{\vec{k}}^\dagger - f_{\vec{k}} \right).
\end{align}
\end{subequations}

\bullet Note, that
\begin{subequations}
\begin{align}
    \Delta k \sum_{i=1}^{N_k} B_k &= \frac{\Delta k}{\sqrt{V}} \sum_{i=1}^{N_k} \sum_{\vec{q}} h(k - |\vec{q}|) O_{\vec{q}} \\
        &\approx \frac{1}{\sqrt{V}} \int \mathrm{d} k \sum_{\vec{q}} \delta(k - |\vec{q}|) O_{\vec{q}} \\
        &= \frac{1}{\sqrt{V}} \sum_{\vec{q}} O_{\vec{q}}.
\end{align}
\end{subequations}
This indicates that the proper starting state $\vec{a}$ for, e.g., the Higgs mode is given by
\begin{equation}
    \vec{a}_i \coloneqq \begin{cases}
        1 & i < N_k \\ 0 & \mathrm{otherwise}
    \end{cases}.
\end{equation}

\bullet For these calculations, we finally require atleast a small screening because the commutators yield otherwise divergent terms.
As an example consider 
$\expec{[ f_{\vec{l}}^\dagger  - f_{\vec{l}} , [H,  f_{\vec{k}} - f_{\vec{k}}^\dagger ]]}$.
One of the resulting terms is
$V( 0 )  \langle f_{ \vec{l} } \rangle \langle f_{ \vec{k} } \rangle$,
where $V(0)$ refers to the Coulomb potential evaluated at $0$.
Because these terms are contributions to individual matrix elements of matrices that we need to manipulated in a non-trivial manner,
we cannot lift nor easily avoid these kinds of singularities and ultimately require a finite screening.
